Simple Contract with State Variables and Functions: typescript Copy code

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract MyContract {
  // State variables
  string public greeting;

  // Constructor
  constructor() {
    greeting = "Hello";
  }

  // Function to set greeting
  function setGreeting(string memory _greeting) public {
    greeting = _greeting;
  }

  // Function to get greeting
  function getGreeting() public view returns (string memory) {
    return greeting;
  }
}

Contract Inheritance: typescript Copy code

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract Ownable {
  address public owner;

  constructor() {
    owner = msg.sender;
  }

  modifier onlyOwner() {
    require(msg.sender == owner, "Only the owner can call this function.");
    _;
  }
}

contract MyContract is Ownable {
  // State variables
  string public message;

  // Function to set message
  function setMessage(string memory _message) public onlyOwner {
    message = _message;
  }
}

Events: csharp Copy code

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract MyContract {
  // Event
  event MessageUpdated(string message);

  // State variable
  string public message;

  // Function to set message and emit event
  function setMessage(string memory _message) public {
    message = _message;
    emit MessageUpdated(_message);
  }
}

Structs and Mappings: csharp Copy code

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract MyContract {
  // Struct
  struct Person {
    string name;
    uint age;
  }

  // Mapping
  mapping(address => Person) public people;

  // Function to add person to mapping
  function addPerson(string memory _name, uint _age) public {
    address creator = msg.sender;
    people[creator] = Person(_name, _age);
  }
}

Error Handling with Revert and Require: scss Copy code

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract MyContract {
  // State variable
  uint public balance;

  // Function to withdraw balance
  function withdraw(uint _amount) public {
    require(_amount <= balance, "Insufficient balance.");
    balance -= _amount;
    if (!_msgSender().send(_amount)) {
        revert("Error: Transaction failed.");
    }
  }
}
